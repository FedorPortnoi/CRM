## Step 2 — Add Twilio HMAC verification helper

In backend/api/controllers/messages.ts, add a helper function ABOVE the webhook handlers:

```typescript
import crypto from 'crypto';

function verifyTwilioSignature(
  authToken: string,
  twilioSignature: string,
  url: string,
  params: Record<string, string>,
): boolean {
  // Sort POST params alphabetically, concatenate key+value pairs to URL
  const sortedKeys = Object.keys(params).sort();
  const data = url + sortedKeys.map(k => k + params[k]).join('');
  const expected = crypto
    .createHmac('sha1', authToken)
    .update(data)
    .digest('base64');
  return crypto.timingSafeEqual(Buffer.from(expected), Buffer.from(twilioSignature));
}
```

## Step 3 — Harden twilioInboundWebhook

Update the function to:
1. Check for TWILIO_AUTH_TOKEN in env. If set, verify the X-Twilio-Signature header.
   If verification fails, return 403. If not set (dev mode), skip verification.
2. After saving the message, return a TwiML response (not JSON) so Twilio doesn't retry.

The updated function should look like:

```typescript
async function twilioInboundWebhook(request: FastifyRequest, reply: FastifyReply): Promise<void> {
  try {
    const authToken = process.env.TWILIO_AUTH_TOKEN;
    if (authToken) {
      const sig = request.headers['x-twilio-signature'] as string | undefined;
      if (!sig) {
        reply.status(403).send('Missing Twilio signature');
        return;
      }
      const url = `${request.protocol}://${request.hostname}${request.url}`;
      const params = request.body as Record<string, string>;
      if (!verifyTwilioSignature(authToken, sig, url, params)) {
        reply.status(403).send('Invalid Twilio signature');
        return;
      }
    }

    const body = request.body as Record<string, unknown>;
    const from = body['From'];
    const messageBody = body['Body'];
    const smsSid = body['SmsSid'];

    const contact = await db.contact.findFirst({
      where: { phone: String(from) },
    });

    if (contact) {
      await db.message.create({
        data: {
          organization_id: contact.organization_id,
          contact_id: contact.id,
          direction: MessageDirection.inbound,
          channel: MessageChannel.sms,
          body: String(messageBody),
          status: MessageStatus.delivered,
          twilio_sid: String(smsSid),
        },
      });
    }
  } catch {
    // silently ignore — Twilio requires 200 regardless
  }

  // Return TwiML (Twilio Markup Language) — empty response means no reply SMS
  reply
    .status(200)
    .header('Content-Type', 'application/xml')
    .send('<?xml version="1.0" encoding="UTF-8"?><Response></Response>');
}
```
