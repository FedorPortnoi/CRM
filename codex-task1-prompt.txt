function authHeaders(token: string): { Authorization: string } {
  return { Authorization: `Bearer ${token}` };

function uniqueSuffix(prefix: string): string {
  return `${prefix}-${Date.now()}-${Math.random().toString(36).slice(2)}`;
}

async function registerOrg(request: APIRequestContext, suffix: string): Promise<AuthOrg> {
  const unique = uniqueSuffix(suffix);
  const res = await request.post('/api/v1/auth/', {
    data: {
      email: `${unique}@example.com`,
      password: 'Password123!',
      name: `User ${suffix}`,
      org_name: `Org ${unique}`,
    },
  });
  expect(res.status()).toBe(201);
  const body = (await res.json()) as RegisterResponse;
  return { token: body.data.token, userId: body.data.user.id };
}

Key backend behaviors confirmed from source code:

POST /api/v1/notifications/register:
  - Schema: { token: z.string().min(1) } — missing token → 400
  - Controller: if (!Expo.isExpoPushToken(token)) → 400 { error: { code: 'INVALID_PUSH_TOKEN', message: '...' } }
  - A non-Expo format string like "not-an-expo-push-token" fails isExpoPushToken → 400 INVALID_PUSH_TOKEN

POST /api/v1/notifications/send:
  - Schema: { user_id: z.string().uuid(), title: z.string().min(1).max(200), body: z.string().min(1).max(1000) }
  - Sending without title field → 400 (Zod schema)

GET /api/v1/contacts?type=customer:
  - Returns contacts matching that type only
  - meta.total reflects ONLY the type-matched count, not all org contacts

POST /api/v1/contacts/bulk-archive:
  - Schema: { contact_ids: z.array(z.string().uuid()).min(1).max(100) }
  - Controller: findMany where { id: { in: uniqueContactIds }, organization_id: orgId, status: { not: 'archived' } }
  - If one contact is already archived, it's excluded from findMany → contacts.length < uniqueContactIds.length → 404 NOT_FOUND
  - If all contact_ids belong to a different org, none are found → contacts.length === 0 < uniqueContactIds.length → 404 NOT_FOUND

GET /api/v1/tasks?due_before=<ISO datetime>:
  - Controller: due_date: { lt: new Date(due_before) }
  - Tasks with due_date >= due_before are excluded
  - due_before must be a valid ISO 8601 datetime string (e.g., "2026-12-31T00:00:00.000Z")

GET /api/v1/deals?status=lost:
  - Controller supports status filter (same as ?status=won tested in 24-coverage G8)
  - Returns only deals with status='lost'
  - Endpoint to mark lost: POST /api/v1/deals/:id/lost with body { reason?: string }

GET /api/v1/tasks?status=in_progress:
  - Returns only in_progress tasks
  - Start a task via POST /api/v1/tasks/:id/start (pending → in_progress)

GET /api/v1/contacts?type=customer&status=active:
  - Both filters applied together
  - Returns contacts that are BOTH type=customer AND status=active

GET /api/v1/tasks?assigned_to=<userId>:
  - Returns only tasks assigned to that specific user
  - A task assigned to userA does NOT appear when filtered by userB's id

PATCH /api/v1/tasks/:id with non-existent id:
  - Controller: findFirst returns null → 404 { error: { code: 'TASK_NOT_FOUND', message: '...' } }
  - Use a random UUID that doesn't exist

POST /api/v1/calendar with end_time <= start_time:
  - Zod schema: CreateEventSchema = EventFieldsSchema.superRefine(validateEventWindow)
  - validateEventWindow: if end_time <= start_time → z.ZodIssueCode.custom on 'end_time' path
  - Returns 400 (Zod validation error, NOT the { error: { code } } envelope — just check status 400)
  - Required fields: title, start_time, end_time (both z.string().datetime())
